#include<stdio.h>
#define size 6
struct Queue
{
   int items[size];
   int f, r;
};

typedef struct Queue QUE;
void Insert( QUE *);
void Delete(QUE *);
void Disp(QUE *);


void main()
{
  int ch;
  QUE Q;
  Q.f= 0; Q.r = -1;
  clrscr();

  {
    printf("enter choice:\n1:Insert\n2:Delete\n3:Display\n");
    scanf("%d", &ch);

       case 1: Insert(&Q); Disp(&Q); break;
       case 2:  Delete(&Q); Disp(&Q); break;
       case 3:   Disp(&Q); break;
       default:  exit(0);
    }
    }
  }
void Insert(QUE *Q) // Inserting in Ascending order
{
  int item, j;
  if(Q->r==size-1)
  {
    printf("Queue Full\n");
    return;
  }
  printf("enter item:");
  scanf("%d", &item);
 // Q->r++;
  j = Q->r;
  while(j>=0 && Q->items[j] > item)
  {

    Q->items[j+1] = Q->items[j];
    j--;
 }
 Q->items[++j] = item;
 Q->r++;
 }
